## Usage

```tsx keepBackground
import { FileTree } from "@/components/nexus-ui/file-tree";
```

```tsx keepBackground
<FileTree
  tree={[
    {
      name: "src",
      children: [
        { name: "app", children: [{ name: "page.tsx" }, { name: "layout.tsx" }] },
        { name: "components", children: [{ name: "header.tsx" }] },
      ],
    },
    { name: "package.json" },
  ]}
/>
```
